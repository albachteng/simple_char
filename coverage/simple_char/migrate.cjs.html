async function runMigrations() {
  let db;
  
  try {
    // Load knex configuration
    const knexConfig = require('./knexfile.cjs');
    const env = process.env.NODE_ENV || 'development';
    const config = knexConfig[env];
    
    console.log(`Environment: ${env}`);
    
    if (typeof config.connection === 'object') {
      console.log(`Host: ${config.connection.host}`);
      console.log(`Database: ${config.connection.database}`);
      console.log(`Port: ${config.connection.port}`);
    } else {
      console.log(`Connection URL: ${config.connection.substring(0, 30)}...`);
    }
    
    // Create Knex instance
    console.log('\nüîÑ Creating database connection...');
    db = knex(config);
    
    // Test connection
    console.log('üîÑ Testing connection...');
    await db.raw('SELECT 1 as test');
    console.log('‚úÖ Connection successful!');
    
    // Run migrations
    console.log('\nüîÑ Running migrations...');
    const [batchNo, log] = await db.migrate.latest();
    
    if (log.length === 0) {
      console.log('‚úÖ Database is already up to date!');
    } else {
      console.log(`‚úÖ Migrated ${log.length} files in batch ${batchNo}:`);
      log.forEach(file => {
        console.log(`  - ${file}`);
      });
    }
    
    console.log('\nüéâ Migration completed successfully!');
    
  } catch (error) {
    console.error('\n‚ùå Migration failed!');
    console.error('Error:', error.message);
    
    if (error.message.includes('relation') && error.message.includes('does not exist')) {
      console.error('\nüí° This might be your first migration.');
      console.error('The error is expected if no tables exist yet.');
    }
    
    process.exit(1);
  } finally {
    if (db) {
      await db.destroy();
      console.log('üîÑ Database connection closed.');
    }
  }
}
 
runMigrations();
